for each server
	-	create a socket
	-	bind it to an address
	-	listen on that socket
	-	accept connecttions
		-	if the client already accepted
			-	use that client
		-	otherwise
			-	create a client for each accepted connection


server:
	-	fd : socket
	-	ip address
	-	port
	-	servername
	-	aliases (other servernames)
	-	root
	-	served locations


location:
	-	allowed methods
	-	redirections
	-	auto index
	-	default served file


client:
	-	fd : socket
	-	request
	-	response
	-	keep alive


request:
	-	msg
	-	method
	-	path
	-	version
	-	headers : pair<string, string> : header, value
	-	body


response:
	-	msg
	-	status code
	-	headers : from the request
	-	body
